    # 3.1 冒泡事件

    打印顺序：我是儿子->我是爸爸-> 我是爷爷

    <body>
  <div id="grandfather">
    <div id="father">
      <div id="son"></div>
    </div>
  </div>
  <script>
  // 1. 获取所有元素
  var grandfather = document.querySelector('#grandfather');
  var father = document.querySelector('#father');
  var son = document.querySelector('#son');
  
  // 2. 添加点击事件
  grandfather.onclick = function(){
  // 3. 在事件处理程序中，打印‘我是谁’
    console.log('我是爷爷');
    
  }
  // 2. 添加点击事件
  father.onclick = function(){
  // 3. 在事件处理程序中，打印‘我是谁’
    console.log('我是爸爸');
    
  }
  // 2. 添加点击事件
  son.onclick = function(){
  // 3. 在事件处理程序中，打印‘我是谁’
    console.log('我是儿子');
    
  }
  </script>
</body>
